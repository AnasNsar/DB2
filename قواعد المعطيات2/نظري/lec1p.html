DECLARE
PI CONSTANT NUMBER(9,7) :=3.1416926;
area NUMBER(14,2); CURSOR rad_cursor is
Select * From radius_vals;
rad_Val rad_cursor%Rowtype;
BEGIN
Open rad_cursor;
Fetch rad_cursor INTO rad_Val;
area:= pi*power(rad_Val.radiuse,2);
Close rad_cursor;
END;
IF area >30 Then
Insert INTO areas values (rad_Val.radius,area);
End IF ;
